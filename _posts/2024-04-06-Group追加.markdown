# commands

```sh
groupadd newgroup
useradd -m newuser1
echo 'from root' > /newfile

usermod -aG newgroup root
usermod -aG newgroup newuser1
chown root:newgroup /newfile
chmod 070 /newfile

su - newuser1
echo 'from newuser1' >> /newfile

cat /newfile
ls -l /newfile 
```
